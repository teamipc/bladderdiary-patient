`?clinic=<5000-char-string>` or `?clinic=<script>` MUST NOT persist the
raw value to localStorage (now IndexedDB per Phase 4). Only alphanumeric-
plus-dash values within a length cap are accepted.

User-locked regex at the gate: `/^[A-Za-z0-9-]{1,32}$/`

### Task 1 — Add the validation regex + guard

In `src/app/[locale]/LandingContent.tsx`:

1. Add a module-level constant at the top of the file (after the
   imports block, around line 33 — between the last import and the
   `function LandingContent() {` declaration). Make it `export const`
   so the test file can import and assert on the regex directly
   (cheaper than running multiple component renders):
   ```ts
   /**
    * Validation for the `?clinic=` URL search param.
    *
    * Rule: alphanumeric + hyphen, 1 to 32 characters. Anything outside
    * this charset (including <script>, multi-thousand-char payloads,
    * URL-encoded entities, whitespace, etc.) is silently rejected —
    * the param is NOT persisted to the diary store. A dev-only
    * console.warn fires for debugging visibility.
    *
    * Locked at gate: see .planning/phases/03-ux-polish-input-validation/03-CONTEXT.md
    * §"STAB-08 — locked details".
    */
   export const CLINIC_CODE_RE = /^[A-Za-z0-9-]{1,32}$/;
   ```

2. Modify the `useEffect` at lines 46-51 from:
   ```tsx
   useEffect(() => {
     const clinic = searchParams.get('clinic');
     if (clinic) {
       setClinicCode(clinic);
     }
   }, [searchParams, setClinicCode]);
   ```
   to:
   ```tsx
   useEffect(() => {
     const clinic = searchParams.get('clinic');
     if (clinic && CLINIC_CODE_RE.test(clinic)) {
       setClinicCode(clinic);
     } else if (clinic && process.env.NODE_ENV !== 'production') {
       // Truncate payload at 100 chars to prevent dev-tools console
       // blow-up on a 5000-char attack string.
       console.warn('Ignored invalid clinicCode:', clinic.slice(0, 100));
     }
   }, [searchParams, setClinicCode]);
   ```

   Critical specifics:
   - Branch order matters: the `else if` only fires when `clinic` is
     truthy AND failed the regex. An ABSENT `?clinic` param (null
     return from `searchParams.get`) falls through both branches —
     no console noise on every page load, only on actual bad input.
   - The `process.env.NODE_ENV !== 'production'` check is statically
     stripped by the Next.js production build, so the `console.warn`
     line is dead-code-eliminated in production. No runtime cost or
     log noise for end users.
   - Do NOT call `setClinicCode(null)` in the else branch. Per
     CONTEXT, the design is **silent reject** — preserve any
     previously-set clinicCode rather than clobbering it.
   - Do NOT add a `try/catch` — `RegExp.prototype.test` cannot throw.
   - Do NOT change the dependency array (still `[searchParams, setClinicCode]`).

```ts
import { describe, it, expect } from 'vitest';
import { CLINIC_CODE_RE } from '@/app/[locale]/LandingContent';

describe('CLINIC_CODE_RE', () => {
  it('accepts IPC-2026', () => {
    expect(CLINIC_CODE_RE.test('IPC-2026')).toBe(true);
  });
  it('accepts single character A', () => {
    expect(CLINIC_CODE_RE.test('A')).toBe(true);
  });
  it('accepts the 32-char maximum (boundary inclusive)', () => {
    expect(CLINIC_CODE_RE.test('A'.repeat(32))).toBe(true);
  });
  it('rejects 33 characters (boundary exclusive)', () => {
    expect(CLINIC_CODE_RE.test('A'.repeat(33))).toBe(false);
  });
  it('rejects <script>', () => {
    expect(CLINIC_CODE_RE.test('<script>')).toBe(false);
  });
  it('rejects empty string', () => {
    expect(CLINIC_CODE_RE.test('')).toBe(false);
  });
  it('rejects underscore (intentional charset narrowing)', () => {
    expect(CLINIC_CODE_RE.test('IPC_2026')).toBe(false);
  });
  it('rejects whitespace', () => {
    expect(CLINIC_CODE_RE.test('IPC 2026')).toBe(false);
  });
  it('rejects URL-encoded payload', () => {
    expect(CLINIC_CODE_RE.test('%3Cscript%3E')).toBe(false);
  });
  it('rejects 5000-char attack payload (length cap)', () => {
    expect(CLINIC_CODE_RE.test('a'.repeat(5000))).toBe(false);
  });
});
```

```ts
describe('clinicCode URL param wiring (integration)', () => {
  it('valid: ?clinic=IPC-2026 → store.clinicCode === "IPC-2026"', async () => {
    mockClinicParam = 'IPC-2026';
    renderLanding();
    // useEffect runs after mount — flush microtasks.
    await Promise.resolve();
    expect(useDiaryStore.getState().clinicCode).toBe('IPC-2026');
  });

  it('valid: ?clinic=A → store.clinicCode === "A"', async () => {
    mockClinicParam = 'A';
    renderLanding();
    await Promise.resolve();
    expect(useDiaryStore.getState().clinicCode).toBe('A');
  });

  it('invalid (too long, 33 chars): does NOT persist; store.clinicCode stays null', async () => {
    mockClinicParam = 'A'.repeat(33);
    renderLanding();
    await Promise.resolve();
    expect(useDiaryStore.getState().clinicCode).toBeNull();
  });

  it('invalid (bad chars, <script>): does NOT persist; store.clinicCode stays null', async () => {
    mockClinicParam = '<script>';
    renderLanding();
    await Promise.resolve();
    expect(useDiaryStore.getState().clinicCode).toBeNull();
  });

  it('invalid (empty string): does NOT persist; store.clinicCode stays null', async () => {
    mockClinicParam = '';
    renderLanding();
    await Promise.resolve();
    // searchParams.get returns '' for ?clinic= — falsy, so the
    // first branch is skipped AND the else-if's truthy check also fails.
    // Store stays at initial null.
    expect(useDiaryStore.getState().clinicCode).toBeNull();
  });

  it('absent: no ?clinic param → store.clinicCode stays null (no behavior change)', async () => {
    mockClinicParam = null;
    renderLanding();
    await Promise.resolve();
    expect(useDiaryStore.getState().clinicCode).toBeNull();
  });

  it('dev-only console.warn fires for invalid (non-production env)', async () => {
    const warnSpy = vi.spyOn(console, 'warn').mockImplementation(() => {});
    // vitest runs with NODE_ENV='test' by default — not 'production' —
    // so the warn branch IS taken.
    mockClinicParam = '<script>';
    renderLanding();
    await Promise.resolve();
    expect(warnSpy).toHaveBeenCalledWith('Ignored invalid clinicCode:', '<script>');
    warnSpy.mockRestore();
  });

  it('console.warn payload truncated to 100 chars for long attack strings', async () => {
    const warnSpy = vi.spyOn(console, 'warn').mockImplementation(() => {});
    const longPayload = 'x'.repeat(5000);
    mockClinicParam = longPayload;
    renderLanding();
    await Promise.resolve();
    expect(warnSpy).toHaveBeenCalledTimes(1);
    const arg = warnSpy.mock.calls[0][1] as string;
    expect(arg.length).toBe(100);
    expect(arg).toBe('x'.repeat(100));
    warnSpy.mockRestore();
  });
});
```
