# Plan 08-02 (OPTIONAL — automated regression guards): New Playwright spec covering C1/C2/C3/C4 regression + expanded physical-CSS guard across ALL components

<objective>
This is the optional Plan 08-02 the planner identified as worth adding per CONTEXT §"Decisions" → "Plan structure": "Optional Plan 08-02 verification spec if the planner identifies new automated guards worth adding." The planner judged YES on this plan for 3 reasons:

1. **Two of the four carry-overs (C1 + C3) are ALREADY observable as failures in the daily walkthrough log** (`walkthrough_findings.md` 2026-05-16 + 2026-05-17 entries: `[summary] Diary tab link to /summary not visible — seed may not have hydrated` and `[deep-flow] [deepLinkSummary] Deep-link redirect race regressed`). This expanded guard catches drift.

    ```ts
    test.describe('Expanded physical-CSS regression guard', () => {
      test('No new physical CSS introduced beyond the explicit allowlist', () => {
        // Walk src/components/ + src/app/ for all .tsx files
        const candidateRoots = ['src/components', 'src/app'];
        const offenders: Array<{ file: string; line: number; text: string }> = [];

        function walk(dir: string) {
          for (const entry of readdirSync(dir)) {
            const full = join(dir, entry);
            const stat = statSync(full);
            if (stat.isDirectory()) {
              walk(full);
            } else if (entry.endsWith('.tsx') || entry.endsWith('.ts')) {
              const lines = readFileSync(full, 'utf8').split('\n');
              lines.forEach((line, idx) => {
                // Skip lines with rtl: variant (intentional RTL adaptation)
                if (/rtl:/.test(line)) return;
                // Match physical-CSS Tailwind classes:
                //   ml-N, mr-N, pl-N, pr-N, left-N, right-N, border-l-, border-r-, text-left, text-right
                // Excluding logical equivalents (ms-, me-, ps-, pe-, start-, end-)
                const physicalCssMatch = line.match(/(^|[^a-zA-Z-])(ml-|mr-|pl-|pr-|left-|right-|border-l-|border-r-|text-left|text-right)[0-9a-zA-Z]/);
                if (!physicalCssMatch) return;
                // Check against allowlist
                const relPath = relative(process.cwd(), full).replace(/\\/g, '/');
                const allowed = PHYSICAL_CSS_ALLOWLIST.some(
                  (entry) => relPath.endsWith(entry.file) && entry.pattern.test(line),
                );
                if (!allowed) {
                  offenders.push({ file: relPath, line: idx + 1, text: line.trim() });
                }
              });
            }
          }
        }

        for (const root of candidateRoots) {
          walk(root);
        }

        if (offenders.length > 0) {
          const report = offenders
            .map((o) => `  ${o.file}:${o.line} — ${o.text}`)
            .join('\n');
          console.log('Physical-CSS violations (NOT in allowlist):');
          console.log(report);
        }

        expect(offenders).toEqual([]);
      });
    });
    ```
