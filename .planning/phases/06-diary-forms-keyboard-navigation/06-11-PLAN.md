       const LOCALES = ['en', 'fr', 'es', 'pt', 'zh', 'ar'] as const;
       type Locale = typeof LOCALES[number];

       const VIEWPORTS = [
         { name: 'mobile-375', width: 375, height: 800 },
         { name: 'desktop-768', width: 768, height: 900 },
         { name: 'desktop-1280', width: 1280, height: 900 },

       const SCREENSHOT_DIR = resolve('test-results/phase6-keyboard');
       mkdirSync(SCREENSHOT_DIR, { recursive: true });

       function localePath(locale: Locale, path: string): string {
         if (locale === 'en') return path;
         return `/${locale}${path}`;
       }

       // Local static-export server. The spec assumes `out/` is built (npm run build).
       // To run: `npm run build && npx --yes serve out -l 4173 --no-clipboard &`
       //   then `PW_TEST_MATCH='phase6-keyboard\.spec\.ts' npx playwright test e2e/phase6-keyboard.spec.ts`

       // Phase 5's executor added the PW_TEST_MATCH hook; Phase 6 reuses it.

         baseURL: BASE_URL,

       ```ts
       for (const locale of LOCALES) {
         for (const viewport of VIEWPORTS) {
           test(`${locale} × ${viewport.name} — drink form renders correctly`, async ({ page }) => {
             // Seed store so /diary/day/1 is reachable
             await page.addInitScript(([key, state]) => {
               window.localStorage.setItem(key, state);
             }, [STORE_KEY, JSON.stringify(buildSeedState({ /* day-1-started-with-wake state */ }))]);

             await page.setViewportSize({ width: viewport.width, height: viewport.height });
             await page.goto(localePath(locale, '/diary/day/1'));

             // Open FAB → drink form
             await page.locator('[data-testid="fab-toggle"]').click();
             await page.locator('[data-testid="fab-action-drink"]').click();

             // Wait for the sheet to be open
             const sheet = page.locator('[data-testid="bottom-sheet"]').last();
             await expect(sheet).toBeVisible();

             // Find the modal card div (the one with role="dialog")
             const dialog = sheet.locator('[role="dialog"][aria-modal="true"]');
             await expect(dialog).toBeVisible();

             // Capture screenshot per locale × viewport
             await page.screenshot({
               path: `${SCREENSHOT_DIR}/drink-form-${locale}-${viewport.name}.png`,
               fullPage: false,
             });

             // Assert max-width by inspecting computed box
             const dialogBox = await dialog.boundingBox();
             expect(dialogBox).not.toBeNull();
             if (viewport.width >= 768) {
               // At md+, modal max-w-3xl = 768px so dialogBox.width <= 768
               expect(dialogBox!.width).toBeLessThanOrEqual(768 + 1);
               // And centered (with 24px p-6 viewport padding accounted for)
               const expectedCenterX = viewport.width / 2;
               const dialogCenterX = dialogBox!.x + dialogBox!.width / 2;
               expect(Math.abs(dialogCenterX - expectedCenterX)).toBeLessThan(50);
             } else {
               // At <768px, modal pinned to viewport bottom — width should be near full viewport
               expect(dialogBox!.width).toBeGreaterThan(viewport.width - 5);
               // y position should be near viewport bottom (sheet pinned at bottom-0)
               expect(dialogBox!.y + dialogBox!.height).toBeGreaterThanOrEqual(viewport.height - 5);
             }
           });
         }
       }
       ```
