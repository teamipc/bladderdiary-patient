JSON-LD extraction pattern (extracts all `<script type="application/ld+json">` blocks from a page):

```ts
async function getJsonLdScripts(page: Page): Promise<unknown[]> {
  return page.evaluate(() => {
    const scripts = Array.from(
      document.querySelectorAll('script[type="application/ld+json"]'),
    );
    return scripts.map((s) => JSON.parse(s.textContent ?? '{}'));
  });
}

function findBreadcrumbList(scripts: unknown[]): { itemListElement: Array<{ position: number; name: string; item: string }> } | null {
  for (const s of scripts) {
    if (typeof s === 'object' && s !== null && (s as { '@type'?: string })['@type'] === 'BreadcrumbList') {
      return s as { itemListElement: Array<{ position: number; name: string; item: string }> };
    }
  }
  return null;
}
```

import { readFileSync } from 'node:fs';

function getPillarTitle(locale: Locale, topic: string): string | null {
  try {
    const content = readFileSync(
      resolve(process.cwd(), 'content', 'articles', locale, topic, '_pillar.mdx'),
      'utf8',

    const m = content.match(/^title:\s*["'](.+?)["']\s*$/m);

    return null;
  }
}
